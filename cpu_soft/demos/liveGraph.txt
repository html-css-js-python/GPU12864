#include "GPU12864.h"

const int graphTop = 10;
const int graphBottom = 58;
const int graphHeight = graphBottom - graphTop;

uint8_t graphData[128];

bool rising = true;
int value = 0;

void setup()
{
  gpuBegin();
  gpuClear();
  
  gpuDrawText(1, 0, "WARNING", FONT_8X8);
  gpuDrawText(1, 8, "Adjust your", FONT_5X7);
  gpuDrawText(1, 16, "GLCD contrast!", FONT_5X7);

  gpuUpdate();

  delay(2500);

  gpuClear();

  for (int i = 0; i < 128; i++)
  {
    graphData[i] = graphBottom;
  }
}

void loop()
{
  for (int i = 0; i < 127; i++)
  {
    graphData[i] = graphData[i + 1];
  }

  if (rising)
  {
    value += 4;
    if (value >= 100)
    {
      value = 100;
      rising = false;
    }
  }
  else
  {
    value -= 4;
    if (value <= 0)
    {
      value = 0;
      rising = true;
    }
  }

  graphData[127] = graphBottom - (value * graphHeight / 100);

  gpuClear();

  gpuDrawText(1, 0, "LIVE TRIANGLE GRAPH", FONT_5X7);

  for (int y = graphTop; y <= graphBottom; y += 12)
  {
    gpuDrawLine(0, y, 127, y, true);
  }

  for (int x = 0; x < 127; x++)
  {
    gpuDrawLine(x, graphData[x], x + 1, graphData[x + 1], true);
  }

  gpuUpdate();

  delay(120);
}
